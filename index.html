1. Copy to clipboard

HTML

    <textarea id="textToCopy" rows="6" cols="60">
        Enter a text to be copied
    </textarea>

    <button class="btn" onclick="copy()">Click to copy</button>

Javascript

    function copy() {
        var copyText = document.getElementById("textToCopy");
        copyText.select();
        document.execCommand("copy");
        alert("Copied text:\n"+copyText.value)
    }
